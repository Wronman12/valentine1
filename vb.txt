    <script>
        let crushName = "";
        let senderName = "";
        let senderWhatsApp = "";
        let noButtonInitialized = false;

        // Check if URL has proposal data
        window.addEventListener('DOMContentLoaded', function() {
            const urlParams = new URLSearchParams(window.location.search);
            const crush = urlParams.get('crush');
            const sender = urlParams.get('sender');
            const whatsapp = urlParams.get('wa');

            if (crush && sender && whatsapp) {
                // Show proposal page
                crushName = decodeURIComponent(crush);
                senderName = decodeURIComponent(sender);
                senderWhatsApp = decodeURIComponent(whatsapp);
                
                document.getElementById('senderPage').classList.add('hidden');
                document.getElementById('proposalPage').classList.remove('hidden');
                document.getElementById('crushNameDisplay').textContent = crushName;
                document.getElementById('senderNameDisplay').textContent = senderName;
                
                // Initialize no button after page loads
                setTimeout(initializeNoButton, 500);
            }

        function createProposal() {
            const crush = document.getElementById('crushNameInput').value.trim();
            const sender = document.getElementById('yourNameInput').value.trim();
            const whatsapp = document.getElementById('whatsappInput').value.trim();

            if (!crush || !sender) {
                alert('â¤ï¸ Please enter both names!');
                return;
            }

            if (!whatsapp || !whatsapp.startsWith('+')) {
                alert('ğŸ“± Please enter your WhatsApp with country code!\n\nExample: +911234567890');
                return;
            }

            // Create shareable link
            const baseUrl = window.location.origin + window.location.pathname;
            const shareLink = `${baseUrl}?crush=${encodeURIComponent(crush)}&sender=${encodeURIComponent(sender)}&wa=${encodeURIComponent(whatsapp)}`;

            // Copy to clipboard
            navigator.clipboard.writeText(shareLink).then(() => {
                alert(`âœ… Link copied to clipboard!\n\nShare this link with ${crush}:\n\n${shareLink}\n\nYou can send it via WhatsApp, Instagram, or any messaging app! ğŸ’•`);
            }).catch(() => {
                // Fallback if clipboard doesn't work
                prompt('ğŸ“‹ Copy this link and send it to your crush:', shareLink);
            });
        }

        function initializeNoButton() {
            if (noButtonInitialized) return;
            noButtonInitialized = true;

            const noBtn = document.getElementById('noBtn');
            const container = document.getElementById('buttonsContainer');
            const yesBtn = document.getElementById('yesBtn');
            
            if (!noBtn || !container) return;

            // Set initial position - place it below yes button
            setTimeout(() => {
                const containerRect = container.getBoundingClientRect();
                const yesRect = yesBtn.getBoundingClientRect();
                
                noBtn.style.left = '50%';
                noBtn.style.transform = 'translateX(-50%)';
                noBtn.style.top = (yesRect.bottom - containerRect.top + 20) + 'px';
            }, 100);

            function moveButton(e) {
                e.preventDefault();
                e.stopPropagation();
                
                const containerRect = container.getBoundingClientRect();
                const btnRect = noBtn.getBoundingClientRect();
                
                let clientX, clientY;
                
                if (e.type.includes('touch')) {
                    if (e.touches && e.touches.length > 0) {
                        clientX = e.touches[0].clientX;
                        clientY = e.touches[0].clientY;
                    } else {
                        return;
                    }
                } else {
                    clientX = e.clientX;
                    clientY = e.clientY;
                }
                
                const btnCenterX = btnRect.left + btnRect.width / 2;
                const btnCenterY = btnRect.top + btnRect.height / 2;
                
                const distance = Math.sqrt(
                    Math.pow(clientX - btnCenterX, 2) + 
                    Math.pow(clientY - btnCenterY, 2)
                );
                
                // Increase detection range for better mobile experience
                if (distance < 150) {
                    const angle = Math.atan2(btnCenterY - clientY, btnCenterX - clientX);
                    const moveDistance = 120;
                    
                    const newX = Math.cos(angle) * moveDistance;
                    const newY = Math.sin(angle) * moveDistance;
                    
                    const currentLeft = btnRect.left - containerRect.left;
                    const currentTop = btnRect.top - containerRect.top;
                    
                    let finalX = currentLeft + newX;
                    let finalY = currentTop + newY;
                    
                    const maxX = containerRect.width - btnRect.width - 10;
                    const maxY = containerRect.height - btnRect.height - 10;
                    
                    finalX = Math.max(10, Math.min(finalX, maxX));
                    finalY = Math.max(10, Math.min(finalY, maxY));
                    
                    noBtn.style.left = finalX + 'px';
                    noBtn.style.top = finalY + 'px';
                    noBtn.style.transform = 'none';
                }
            }
            
            // Mouse events for desktop
            container.addEventListener('mousemove', moveButton);
            noBtn.addEventListener('mouseenter', moveButton);
            
            // Touch events for mobile
            container.addEventListener('touchstart', moveButton, { passive: false });
            container.addEventListener('touchmove', moveButton, { passive: false });
            noBtn.addEventListener('touchstart', moveButton, { passive: false });
        }
